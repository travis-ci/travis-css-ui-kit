import clsx from 'clsx';

<Meta
  title="Components/Button"
  argTypes={{
    label: {
      control: 'text',
      description: 'Button label',
    },
    variant: {
      control: {
        type: 'select',
        options: ['default', 'outline', 'light-outline', 'filled', 'link'],
      },
      description: 'Button variant',
    },
    color: {
      control: {
        type: 'select',
        options: ['default', 'primary', 'secondary', 'bright'],
      },
      description: 'Color variant of a button',
    },
    size: {
      control: {
        type: 'select',
        options: ['default', 'xs', 'sm', 'lg'],
      },
      description: 'Button size',
    },
    ripple: {
      control: {
        type: 'select',
        options: ['default', 50, 100, 200, 300, 400, 500],
      },
      description: 'Size of the ripple effect',
    },
    hover: {
      control: 'boolean',
      description: 'Enable hover effect',
    },
    focus: {
      control: 'boolean',
      description: 'Enable focus effect',
    },
  }}
/>

export const Template = args => {
  const classes = clsx({
    'tci-button': true,
    'tci-button--enable-focus': !!args.focus,
    'tci-button--disable-hover': !args.hover,
    'tci-button--ripple-50': args.ripple === 50,
    'tci-button--ripple-100': args.ripple === 100,
    'tci-button--ripple-200': args.ripple === 200,
    'tci-button--ripple-300': args.ripple === 300,
    'tci-button--ripple-400': args.ripple === 400,
    'tci-button--ripple-500': args.ripple === 500,
    'tci-button--outline': args.variant === 'outline',
    'tci-button--light-outline': args.variant === 'light-outline',
    'tci-button--filled': args.variant === 'filled',
    'tci-button--link': args.variant === 'link',
    'tci-button--primary': args.color === 'primary',
    'tci-button--secondary': args.color === 'secondary',
    'tci-button--bright': args.color === 'bright',
    'tci-button--xs': args.size === 'xs',
    'tci-button--sm': args.size === 'sm',
    'tci-button--lg': args.size === 'lg',
  });
  return `
    <button class="${classes}">
      ${args.label}
    </button>
  `;
};
